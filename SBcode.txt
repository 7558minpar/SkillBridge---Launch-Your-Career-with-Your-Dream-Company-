        function initApp() {
            // Check if user is already logged in
            const savedUser = localStorage.getItem('skillbridge_user');
            if (savedUser) {
                const userData = JSON.parse(savedUser);
                state.currentUser = userData.username;
                state.userSkills = userData.skills || [];
                state.companyInterests = userData.companyInterests || [];
                state.points = userData.points || 0;
                state.achievements = userData.achievements || [];
                showMainContent();
                renderProfilePage();
            }
            
            // Load leaderboard
            loadLeaderboard();
            
            // Update points display
            updatePointsDisplay();
            
            // Initialize suggested skills
            renderSuggestedSkills();
            
            }
